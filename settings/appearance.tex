% Generate clickable links in PDF documents
\usepackage[
 pdfpagelabels,
 hypertexnames=false,  % Make link-names unique -- even in corner cases of subequations
 bookmarksopen=true,
 bookmarksopenlevel=1,  % PDF bookmarks are opened to the first level by default
 colorlinks=true,
 urlcolor=Blue,
 citecolor=Green,
 linkcolor=BrickRed
]{hyperref}
\pdfstringdefDisableCommands{\let\boldmath\relax} % allow \boldmath in section titles

% Named references with the \cref command
\usepackage[
 capitalise,  % "Theorem 3.1" instead of "theorem 3.1"
 nameinlink,  % Makes _Theorem 3.1_ a hyperref link instead of Theorem _3.1_
 noabbrev,  % "Theorem 3.1" instead of "Thm. 3.1"
]{cleveref}

% Named colors
\usepackage[dvipsnames]{xcolor}

% Captions and subcaptions for figures/tables
\usepackage[
 labelfont=bf,  % bold labels (eg. for "Figure 1")
]{caption}
\usepackage{subcaption}

% Tables
\usepackage{booktabs}  % Great tables
\usepackage{multirow}
\usepackage{array}  % fine grained control over tabular spacing
\usepackage{makecell} % Line breaks inside cells
\usepackage{siunitx} % Provides alignment at . for numerical columns using 's'
